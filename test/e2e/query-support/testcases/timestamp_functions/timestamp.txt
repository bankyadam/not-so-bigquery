--SQL--
SELECT TIMESTAMP("2008-12-25 15:30:00+00") AS timestamp_utc1,
       TIMESTAMP("2008-12-25 15:30:00 UTC") AS timestamp_utc2,
       TIMESTAMP(DATETIME "2008-12-25 15:30:00") AS timestamp_datetime,
       TIMESTAMP(DATE "2008-12-25") AS timestamp_date,
       TIMESTAMP("2008-12-25 15:30:00", "America/Los_Angeles") AS timestamp_wtz;

--RESULT--

| timestamp_utc1           | timestamp_utc2           | timestamp_datetime       | timestamp_date           | timestamp_wtz            |
