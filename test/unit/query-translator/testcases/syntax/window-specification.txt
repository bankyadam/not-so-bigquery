--INPUT--
SELECT
    COUNT(*) OVER (PARTITION BY MOD(x, 3)) AS count_star,
    AVG(x) OVER (ORDER BY x ROWS BETWEEN 1 PRECEDING AND CURRENT ROW) AS avg1,
    AVG(x) OVER (RANGE CURRENT ROW) AS avg2,
    AVG(x) OVER (ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING) AS avg3,
    AVG(x) OVER (ROWS BETWEEN 1 PRECEDING AND 2 FOLLOWING) AS avg4
--EXPECT--
SELECT
    COUNT(*) OVER (PARTITION BY MOD(x, 3)) AS count_star,
    AVG(x) OVER (ORDER BY x NULLS FIRST ROWS BETWEEN 1 PRECEDING AND CURRENT ROW) AS avg1,
    AVG(x) OVER (RANGE CURRENT ROW) AS avg2,
    AVG(x) OVER (ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING) AS avg3,
    AVG(x) OVER (ROWS BETWEEN 1 PRECEDING AND 2 FOLLOWING) AS avg4
