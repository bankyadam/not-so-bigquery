--INPUT--
SELECT
    TIMESTAMP '2016-12-25 05:30:00+07' AS tswtz,
    TIMESTAMP '2021-05-01 13:13:13' AS ts,
    DATE '2021-05-02' AS d,
    DATETIME '2021-05-03 13:13:13' AS dt,
    TIME '13:13:13' AS t

--EXPECT--
SELECT
    TIMESTAMP WITH TIME ZONE '2016-12-25 05:30:00+07' AS "tswtz",
    TIMESTAMP WITH TIME ZONE '2021-05-01 13:13:13' AS "ts",
    DATE '2021-05-02' AS "d",
    TIMESTAMP WITH TIME ZONE '2021-05-03 13:13:13' AS "dt",
    TIME '13:13:13' AS "t"
